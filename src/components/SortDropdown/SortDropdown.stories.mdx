import { Meta, Story, Canvas } from '@storybook/addon-docs';

import SortDropdown from './SortDropdown';

<Meta
	title='Core/SortDropdown'
	component={SortDropdown}
	args={{
		options: ['recentlyadded', 'relevance'],
	}}
/>

	<Story name='Basic'>
		{props => {
			const [value, setValue] = React.useState('relevance');

				setValue(value);
			}, []);
			return <SortDropdown {...props} onSort={handleSort} value={value} width='350px' />;
		}}
	</Story>
